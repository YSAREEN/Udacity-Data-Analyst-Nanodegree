#To analyze the increase and descrease in goals for teams over a period of time .'Goal data' for each team in a season we join the Team table, match table and league table
df_goal = pd.read_sql_query("""select sum(goals) Goals_scored, season, team_name, league_name from (
select ht.team_long_name team_name, lg.name league_name, sum(home_team_goal) goals, m.season
from match m
	join team ht on m.home_team_api_id = ht.team_api_id
	join league lg on m.league_id = lg.id
group by m.season, lg.name, ht.team_long_name
UNION
select aw.team_long_name,lg.name, sum(away_team_goal) goals , m.season
from match m
	join team aw on m.away_Team_api_id = aw.team_api_id
	join league lg on m.league_id = lg.id
group by m.season, lg.name, aw.team_long_name
) a
group by season,league_name, team_name
order by league_name asc, sum(goals)""",cnx)
